# FOR SSD "/sd(*=whatever the drive is)" like /sda for drive 'a'
# FOR NVME "/nvme(*)n(*)" like /nvme1n1 for drive 1
# For WiFi use iwctl

00 setfont ter-132b
00 timedatectl set-ntp true
00 pacman -Sy archlinux-keyring

01 lsblk to list disks
02 cfdisk /dev/sda
	First- 1GB; type:EFI System for Boot
	Second- For Swap: Min 4GB, match how much ram have; type:Linux swap
	Third- Main: Whatever is left
	Write
03 mkfs.fat -F32 /dev/sda1
04 mkswap /dev/sda2
05 swapon /dev/sda2
06 mkfs.btrfs -L 'archroot' /dev/sda3; Your Label is whatever you want
07 mount /dev/sda3 /mnt
08 btrfs subvolume create /mnt/@
09 btrfs subvolume create /mnt/@home
10 btrfs subvolume create /mnt/@log
11 btrfs subvolume create /mnt/@cache
12 !!  btrfs subvolume create /mnt/@snapshots
13
14 umount /mnt
15 mount -o noatime,compress=zstd,space_cache=v2,subvol=@ /dev/sda3 /mnt
16 mkdir -p /mnt/{boot/efi,home,.snapshots,var/log,var/cache}
17 mount -o noatime,compress=zstd,space_cache=v2,subvol=@home /dev/sda3 /mnt/home
18 mount -o noatime,compress=zstd,space_cache=v2,subvol=@log /dev/sda3 /mnt/var/log
19 mount -o noatime,compress=zstd,space_cache=v2,subvol=@cache /dev/sda3 /mnt/var/cache
20   !!mount -o noatime,compress=zstd,space_cache=v2,subvol=@snapshots /dev/sda3 /mnt/.snapshots
21 mount /dev/sda1 /mnt/boot/efi
22 pacstrap -K /mnt base base-devel linux linux-lts linux-firmware \
networkmanager grub efibootmgr git sddm plasma-meta konsole nano (if wanted:zram-generator)
23 genfstab -U /mnt >> /mnt/etc/fstab


24 arch-chroot /mnt
25 ln -sf /usr/share/zoneinfo/America/Eastern /etc/localtime
 hwclock --systohc
 nano /etc/locale.gen
 uncommant en_US
26 locale-gen
27 echo "LANG=en_US.UTF-8" >> /etc/locale.conf
28 echo "YOUR_Hostname" >> /etc/hostname
29 passwd 'ROOT_PASSWORD'
30 useradd -m -G wheel 'USER_NAME'
31 passwd 'USER_NAME'
32 EDITOR=nano visudo
33 uncommant #%wheel ALL=(ALL:ALL) ALL
 nano /etc/systemd/zram-generator.conf
	[zram0]
	zram-size = ram / 2
	compression-algorithm = zstd
	swap-priority = 100
	mount-point = /dev/zram0
34 systemctl enable NetworkManager sddm /dev/zram0
35 grub-install --target=x86_64-efi --efi-directory=/boot/efi --bootloader-id=GRUB
36 grub-mkconfig -o /boot/grub/grub.cfg
 exit
37 unmount -R /mnt
38 reboot
